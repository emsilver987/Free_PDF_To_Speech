PDF to Speech - JavaScript Integration Examples

Example 1: Single File Conversion

Convert a PDF to speech asynchronously and track the job.

async function convertSingle() {
  const file = document.getElementById('singleFile').files[0];
  if (!file) return alert('Select a file');
  
  const formData = new FormData();
  formData.append('file', file);
  formData.append('engine', 'gtts');
  formData.append('speed', 1.0);
  formData.append('async', true);
  
  const response = await fetch('http://localhost:5000/api/convert', {
    method: 'POST',
    body: formData
  });
  
  const data = await response.json();
  console.log('Job ID:', data.job_id);
  console.log('Status:', data.status);
  
  // Poll for completion
  await waitForCompletion(data.job_id);
}

Example 2: Batch Processing

Convert multiple PDFs at once.

async function convertBatch() {
  const files = document.getElementById('batchFiles').files;
  if (!files.length) return alert('Select files');
  
  const formData = new FormData();
  for (let file of files) {
    formData.append('files', file);
  }
  formData.append('engine', 'gtts');
  formData.append('speed', 1.2);
  
  const response = await fetch('http://localhost:5000/api/batch', {
    method: 'POST',
    body: formData
  });
  
  const data = await response.json();
  console.log('Jobs queued:', data.job_count);
  console.log('Job IDs:', data.job_ids);
}

Example 3: Voice Selection

Dynamically load and select voices.

async function loadVoices() {
  const engine = document.getElementById('engineSelect').value;
  
  const response = await fetch(
    `http://localhost:5000/api/voices?engine=${engine}`
  );
  
  const data = await response.json();
  console.log('Available voices:', data.voices);
  
  // Populate select dropdown
  const select = document.getElementById('voiceSelect');
  select.innerHTML = data.voices
    .map(v => ``)
    .join('');
}

Example 4: Job Status Polling

Check job status and download when ready.

async function checkStatus() {
  const jobId = document.getElementById('jobId').value;
  if (!jobId) return alert('Enter job ID');
  
  const response = await fetch(
    `http://localhost:5000/api/jobs/${jobId}`
  );
  
  if (response.ok) {
    const job = await response.json();
    console.log('Status:', job.status);
    console.log('Created:', job.created_at);
    console.log('Completed:', job.completed_at);
    
    if (job.status === 'completed') {
      console.log('Ready to download:', job.output_file);
    }
  } else {
    console.error('Job not found');
  }
}

async function downloadResult() {
  const jobId = document.getElementById('jobId').value;
  const response = await fetch(
    `http://localhost:5000/api/download/${jobId}`
  );
  
  const blob = await response.blob();
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = `audio_${jobId}.mp3`;
  a.click();
}

Example 5: Complete Workflow

Full workflow: convert, wait, and download.

async function runWorkflow() {
  // 1. Upload and convert
  const file = document.getElementById('workflowFile').files[0];
  const formData = new FormData();
  formData.append('file', file);
  formData.append('engine', 'gtts');
  formData.append('async', true);
  
  const convertRes = await fetch('http://localhost:5000/api/convert', {
    method: 'POST',
    body: formData
  });
  const job = await convertRes.json();
  const jobId = job.job_id;
  
  // 2. Poll until complete
  let status = 'processing';
  while (status !== 'completed' && status !== 'failed') {
    const statusRes = await fetch(
      `http://localhost:5000/api/jobs/${jobId}`
    );
    const jobData = await statusRes.json();
    status = jobData.status;
    
    console.log(`Status: ${status}`);
    await new Promise(r => setTimeout(r, 2000)); // Wait 2s
  }
  
  if (status === 'completed') {
    // 3. Download
    const downloadRes = await fetch(
      `http://localhost:5000/api/download/${jobId}`
    );
    const blob = await downloadRes.blob();
    const url = URL.createObjectURL(blob);
    
    const a = document.createElement('a');
    a.href = url;
    a.download = file.name.replace('.pdf', '.mp3');
    a.click();
  }
}
